import express from "express"; import fetch from "node-fetch"; const app = express(); app.use(express.json()); const PORT = process.env.PORT || 3000; const OPENAI_API_KEY = process.env.OPENAI_API_KEY; // ---------- FRONTEND ---------- app.get("/", (req, res) => { res.send(` Currículo com IA

Seu currículo feito sob medida para cada vaga

Cole seu currículo e a descrição da vaga. A IA adapta o texto para passar em ATS e chamar atenção do recrutador. 

Aguardando...
MVP • Currículo com IA • Produto validável
`); }); // ---------- BACKEND + IA ---------- app.post("/gerar", async (req, res) => { const { curriculo, vaga } = req.body; if (!curriculo || !vaga) { return res.json({ resultado: "Preencha o currículo e a vaga." }); } const prompt = ` Você é um recrutador sênior especialista em ATS. Reescreva o currículo abaixo para a vaga descrita. Regras: - Use palavras-chave da vaga - Linguagem profissional - Destaque resultados - Não invente experiências CURRÍCULO: ${curriculo} VAGA: ${vaga} `; try { const response = await fetch("https://api.openai.com/v1/chat/completions", { method: "POST", headers: { "Content-Type": "application/json", Authorization: \`Bearer \${OPENAI_API_KEY}\` }, body: JSON.stringify({ model: "gpt-4o-mini", messages: [{ role: "user", content: prompt }] }) }); const data = await response.json(); res.json({ resultado: data.choices[0].message.content }); } catch { res.json({ resultado: "Erro ao gerar currículo." }); } }); // ---------- SERVIDOR ---------- app.listen(PORT, () => { console.log("Servidor rodando em http://localhost:" + PORT); });
